```python
import requests
import json
from jsonpath import jsonpath

def fetch_movies(page=1, page_size=10):
    """获取电影列表"""
    url = "https://api.example.com/movies"
    params = {"page": page, "page_size": page_size}

    response = requests.get(url, params=params, timeout=10)
    data = response.json()

    if data.get('code') != 200:
        raise Exception(f"API错误: {data.get('message')}")

    return data

def get_movie_list(data):
    """使用JSONPath提取电影列表"""
    movies = jsonpath(data, '$.data.movies[*]')
    return movies or []
```

```python
def safe_get_movie_detail(movie):
    """安全获取电影详情"""
    return {
        'id': movie.get('id', ''),
        'title': movie.get('title', ''),
        'rating': movie.get('rating', 0.0),
        'director': movie.get('director', ''),

        'genre': movie.get('genre', []),
        'release_date': movie.get('release_date', ''),
        'duration': movie.get('duration', 0),
        'votes': movie.get('votes', 0),
        'top_comment': get_top_comment(movie),
    }

def get_top_comment(movie):
    """安全获取评论（处理嵌套异常）"""
    comments = movie.get('comments', [])
    if not comments:
        return None

    # 使用jsonpath更安全
    top = jsonpath(movie, '$.comments[0]')
    if top:
        return {
            'user': top[0].get('user', ''),
            'score': top[0].get('score', 0),
            'content': top[0].get('content', '')
        }
    return None
```

**3. 数据清洗**

```python
import re

def clean_rating(rating):
    """清洗评分数据"""
    if isinstance(rating, (int, float)):
        return round(float(rating), 1)
    rating_str = str(rating)
    match = re.search(r'\d+\.?\d*', rating_str)
    return float(match.group()) if match else 0.0

def clean_duration(duration):
    """清洗时长数据（分钟）"""
    if isinstance(duration, int):
        return duration
    match = re.search(r'\d+', str(duration))
    return int(match.group()) if match else 0
```

**4. 数据分析**

```python
from collections import Counter

def analyze_movies(movies):
    """分析电影数据"""
    if not movies:
        return {}

    ratings = [m.get('rating', 0) for m in movies]

    all_genres = []
    for movie in movies:
        all_genres.extend(movie.get('genre', []))

    genre_counter = Counter(all_genres)

    return {
        'total_count': len(movies),
        'avg_rating': round(sum(ratings) / len(ratings), 2),
        'max_rating': max(ratings),
        'min_rating': min(ratings),
        'top_genres': genre_counter.most_common(5),
    }
```

```python
import csv

def export_to_json(movies, filename):
    """导出为JSON"""
    with open(filename, 'w', encoding='utf-8') as f:
        json.dump(movies, f, ensure_ascii=False, indent=2)

def export_to_csv(movies, filename):
    """导出为CSV"""
    if not movies:
        return

    keys = movies[0].keys()

    with open(filename, 'w', encoding='utf-8', newline='') as f:
        writer = csv.DictWriter(f, fieldnames=keys)
        writer.writeheader()
        writer.writerows(movies)
```
